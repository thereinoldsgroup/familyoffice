-- ── EQUITY & CAP TABLE ── create table equity_classes ( id text primary key, company_id text references companies(id) on delete cascade, class_name text not null, -- Common, Series A, Series B, Options, Warrants, etc. class_type text default 'common', -- common, preferred, options, warrants, convertible_note authorized_shares bigint default 0, issued_shares bigint default 0, par_value numeric(18,6) default 0, liquidation_preference numeric(18,2) default 0, conversion_ratio numeric(10,6) default 1, is_voting boolean default true, dividend_rate numeric(8,4) default 0, notes text, created_at timestamptz default now() ); create table cap_table_entries ( id text primary key, company_id text references companies(id) on delete cascade, equity_class_id text references equity_classes(id) on delete cascade, holder_name text not null, holder_type text default 'individual', -- individual, entity, trust, esop shares bigint default 0, original_issue_price numeric(18,6) default 0, current_value_per_share numeric(18,6) default 0, issue_date date, vesting_start date, vesting_months integer default 0, cliff_months integer default 0, is_fully_vested boolean default false, certificate_number text, notes text, created_at timestamptz default now(), updated_at timestamptz default now() ); create table equity_transactions ( id text primary key, company_id text references companies(id) on delete cascade, transaction_type text not null, -- issuance, transfer, repurchase, conversion, exercise, split transaction_date date not null, from_holder text, to_holder text, equity_class_id text references equity_classes(id) on delete set null, shares bigint default 0, price_per_share numeric(18,6) default 0, total_consideration numeric(18,2) default 0, description text, journal_entry_id text references journal_entries(id) on delete set null, created_at timestamptz default now() ); -- ── LOANS & DEBT ── create table loan_accounts ( id text primary key, company_id text references companies(id) on delete cascade, loan_name text not null, lender_name text, loan_type text default 'term_loan', -- term_loan, revolving, mortgage, equipment, sba, line_of_credit original_principal numeric(18,2) default 0, current_balance numeric(18,2) default 0, interest_rate numeric(8,6) default 0, -- annual rate as decimal e.g. 0.065 rate_type text default 'fixed', -- fixed, variable, prime_plus origination_date date, maturity_date date, payment_frequency text default 'monthly', -- monthly, quarterly, annually, interest_only monthly_payment numeric(18,2) default 0, collateral text, gl_principal_account_id text references accounts(id) on delete set null, gl_interest_account_id text references accounts(id) on delete set null, is_active boolean default true, notes text, created_at timestamptz default now(), updated_at timestamptz default now() ); create table loan_payments ( id text primary key, loan_id text references loan_accounts(id) on delete cascade, company_id text references companies(id) on delete cascade, payment_date date not null, payment_number integer, beginning_balance numeric(18,2) default 0, payment_amount numeric(18,2) default 0, principal_portion numeric(18,2) default 0, interest_portion numeric(18,2) default 0, ending_balance numeric(18,2) default 0, is_posted boolean default false, journal_entry_id text references journal_entries(id) on delete set null, notes text, created_at timestamptz default now() ); -- ── HR & BENEFITS ── create table benefit_plans ( id text primary key, company_id text references companies(id) on delete cascade, plan_name text not null, plan_type text not null, -- health, dental, vision, life, disability, retirement, fsa, hsa, other carrier text, plan_number text, effective_date date, renewal_date date, employee_premium_monthly numeric(18,2) default 0, employer_premium_monthly numeric(18,2) default 0, employee_family_premium numeric(18,2) default 0, employer_family_premium numeric(18,2) default 0, is_active boolean default true, notes text, created_at timestamptz default now() ); create table employee_benefit_elections ( id text primary key, employee_id text references employees(id) on delete cascade, plan_id text references benefit_plans(id) on delete cascade, company_id text references companies(id) on delete cascade, coverage_tier text default 'employee', -- employee, employee_spouse, employee_children, family employee_contribution numeric(18,2) default 0, employer_contribution numeric(18,2) default 0, effective_date date, termination_date date, is_active boolean default true, created_at timestamptz default now(), unique(employee_id, plan_id) ); create table hr_events ( id text primary key, employee_id text references employees(id) on delete cascade, company_id text references companies(id) on delete cascade, event_type text not null, -- hire, termination, promotion, demotion, raise, leave, return, review, warning, transfer event_date date not null, description text, previous_value text, -- e.g. previous salary, previous title new_value text, -- e.g. new salary, new title performed_by text, notes text, created_at timestamptz default now() ); create table pto_balances ( id text primary key, employee_id text references employees(id) on delete cascade, company_id text references companies(id) on delete cascade, pto_type text default 'vacation', -- vacation, sick, personal, bereavement, jury_duty accrual_rate_per_period numeric(8,4) default 0, -- hours per pay period current_balance_hours numeric(10,2) default 0, ytd_accrued numeric(10,2) default 0, ytd_used numeric(10,2) default 0, max_carryover_hours numeric(10,2) default 0, as_of_date date, unique(employee_id, pto_type) );
